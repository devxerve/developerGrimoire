# Pseudocodigo

```java title="Pseudocodigo"
usuario{
	float pesoUsuario = x
	int actividadFisica = y
	int altura = z // Altura en cm
	float aguaDeseada = 1,5F
}
// Se quiere modificar peso
pesoUsuario = modificarPeso(pesoUsuario)
// Se quiere modificar Altura
altura = modificarAltura(altura)
// Se quiere modificar actividad fisica
actividadFisica = modificarActividad(actividadFisica)
// Se quiere modificar aguaDeseada
aguaDeseada = modificarAguaDeseada(aguaDeseada);

modificarPeso(peso){
	this.peso = peso
}

modificarAltura(altura){
	this.altura = altura
}

modificarActividad(actividad){
	this.actividadFisica = actividad
}

modificarAguaDeseada(aguaDeseada){
	this.aguaDeseada = aguaDeseada;
	
}
```
